$$ u_{t} = u_{xx}\\ 0 < x < 1 \\ \textbf{Conditions :} \\ u(x,0) = sin \Pi x\\ u(0,t) = u(1,t) = 0 $$

def forward_march(u, r):
    if r <= 1e-10:
        return u
    N = len(u)
    u_ = np.zeros(N, np.float64)
    for i in range(1, N-1):
        u_[i] = r*u[i-1] + (1-2*r)*u[i] + r*u[i+1]
    return u_
